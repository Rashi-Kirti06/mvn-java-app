0. Architecture (What you’re building)
Developer → GitHub → Jenkins → Maven Build → Docker → Docker Hub → Kubernetes → Live App
🔹 1. Create Java Maven Application

vs terminal- 
mvn archetype:generate -DgroupId=com.example -DartifactId=java-app -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false
cd java-app
🔹 Modify App.java
package com.example;

public class App {
    public static void main(String[] args) {
        System.out.println("Hello from CI/CD Java App 🚀");
    }
}
🔹 2. Build using Maven
mvn clean package

👉 Output:

target/java-app-1.0-SNAPSHOT.jar
🔹 3. Create Dockerfile

📍 Inside project root

FROM openjdk:17
WORKDIR /app
COPY target/java-app-1.0-SNAPSHOT.jar app.jar
CMD ["java", "-jar", "app.jar"]
🔹 4. Push to GitHub

git remote add origin https://github.com/yourusername/java-app.git

🔹 9. Create Jenkins Pipeline
✅ Jenkinsfile
pipeline {
    agent any

    tools {
        maven 'Maven'
        jdk 'JDK17'
    }

    environment {
        DOCKER_IMAGE = "yourusername/java-app"
    }

    stages {

        stage('Clone Code') {
            steps {
                git branch: 'main',
                url: 'https://github.com/yourusername/java-app.git'
            }
        }

        stage('Build with Maven') {
            steps {
                bat 'mvn clean package'
            }
        }

        stage('Build Docker Image') {
            steps {
                bat 'docker build -t %DOCKER_IMAGE% .'
            }
        }

        stage('Push Docker Image') {
            steps {
                withCredentials([usernamePassword(
                    credentialsId: 'dockerhub',
                    usernameVariable: 'USER',
                    passwordVariable: 'PASS')]) {

                    bat 'docker login -u %USER% -p %PASS%'
                    bat 'docker push %DOCKER_IMAGE%'
                }
            }
        }

        stage('Deploy to Kubernetes') {
            steps {
                withCredentials([file(
                    credentialsId: 'kuberconfig',
                    variable: 'KUBECONFIG')]) {

                    bat '''
                    set KUBECONFIG=%KUBECONFIG%
                    kubectl apply -f deployment.yaml
                    '''
                }
            }
        }
    }
}
